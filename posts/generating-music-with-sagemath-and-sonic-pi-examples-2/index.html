%python
import numpy as np
import OSC
from time import sleep

addr = ('localhost', 4559)
cl = OSC.OSCClient()
cl.connect(addr)

def send_message(url, *args):
    msg = OSC.OSCMessage()
    msg.setAddress(url)
    for l in args:
        msg.append(l)
    cl.send(msg)

c = 1.25
v = np.array([0, 0.4], dtype=float)
b = 0.25
lst = [[-12, -5, 0], [-8, -3, 4], [-5, -13, 2], [-3, 0, -8], [-2, -12, -8], [-3, 0, -8], [-5, -1, 2], [-8, -3, -20]]
li = np.array(lst, dtype=int)

def bld(l1, rep):
    for r in range(rep):
        for l in l1:
            for p in v/c:
                dur = np.array([p, p+b], dtype=float)/c
                send_message("/trigger/play", l, dur)
                sleep(dur[1])

li2 = [[li, 1], [li + 5, 1], [li, 2], [li + 5, 2], [li, 2], [li + 7, 1], [li + 5, 1], [li, 1], [li + 7, 1]]

for l in li2:
    bld(l[0], l[1])
